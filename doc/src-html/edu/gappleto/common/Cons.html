006package edu.gappleto.common;
007
008import edu.wpi.first.wpilibj.DriverStationLCD;
009import edu.wpi.first.wpilibj.DriverStationLCD.Line;
010import edu.wpi.first.wpilibj.command.Subsystem;
011
012import edu.gappleto.common.Logger;
013
014/**
015 *
016 * @author gappleto97
017 */
018public class Cons extends Subsystem
019{
020    public Cons(DriverStationLCD dStation)
021    {
022        Station = dStation;
023    }
024    
025    private int currentLine = 1;
026    
027    private DriverStationLCD Station;
028    private Line    Line1 = DriverStationLCD.Line.kUser1,
029                    Line2 = DriverStationLCD.Line.kUser2,
030                    Line3 = DriverStationLCD.Line.kUser3,
031                    Line4 = DriverStationLCD.Line.kUser4,
032                    Line5 = DriverStationLCD.Line.kUser5,
033                    Line6 = DriverStationLCD.Line.kUser6;
034    
035    Logger Log = new Logger();
036    
037    public void update()
038    {
039        Station.updateLCD();
040    }
041    
042    public void clear()
043    {
044        Log.println("clearing");
045        Station.clear();
046        Log.println("Updating.");
047        Station.updateLCD();
048        currentLine = 1;
049    }
050    
051    public void println(String input, int Line)
052    {
053        Log.println("Determining line");
054        Log.println("Line is "+Line);
055        if (Line % 6 == 1)
056        {
057            Station.println(Line1, 0, input);
058        }
059        else if (Line % 6 == 2)
060        {
061            Station.println(Line2, 0, input);
062        }
063        else if (Line % 6 == 3)
064        {
065            Station.println(Line3, 0, input);
066        }
067        else if (Line % 6 == 4)
068        {
069            Station.println(Line4, 0, input);
070        }
071        else if (Line % 6 == 5)
072        {
073            Station.println(Line5, 0, input);
074        }
075        else if (Line % 6 == 0)
076        {
077            Station.println(Line6, 0, input);
078        }
079        Log.println("Written");
080        currentLine = Line + 1;
081        if (currentLine > 6)
082        {
083            currentLine = currentLine % 6;
084        }
085        Log.println("Updating screen");
086        Station.updateLCD();
087    }
088    
089    public void println(String input)
090    {
091        if (currentLine % 6 == 1)
092        {
093            Station.println(Line1, 0, input);
094        }
095        else if (currentLine % 6 == 2)
096        {
097            Station.println(Line2, 0, input);
098        }
099        else if (currentLine % 6 == 3)
100        {
101            Station.println(Line3, 0, input);
102        }
103        else if (currentLine % 6 == 4)
104        {
105            Station.println(Line4, 0, input);
106        }
107        else if (currentLine % 6 == 5)
108        {
109            Station.println(Line5, 0, input);
110        }
111        else if (currentLine % 6 == 0)
112        {
113            Station.println(Line6, 0, input);
114        }
115        currentLine++;
116        if (currentLine > 6)
117        {
118            currentLine = currentLine % 6;
119        }
120        Station.updateLCD();    
121    }
