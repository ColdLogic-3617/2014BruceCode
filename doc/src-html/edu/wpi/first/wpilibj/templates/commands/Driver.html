010public class Driver extends CommandBase
011{
012
013    int Func;
014
015    public Driver()
016    {
017        // Use requires() here to declare subsystem dependencies
018        // eg. requires(chassis);
019        requires(DriveTrain);
020        /*
021         * When ititializing Driver(), always include an integer.
022         * You can see possible values values in the RobotMap.
023         * --Gabe
024         */
025    }
026
027    // Called just before this Command runs the first time
028    protected void initialize()
029    {
030    }
031
032    // Called repeatedly when this Command is scheduled to run
033    protected void execute()
034    {
035        TimeSet();
036        DriveTrain.JoyDrive();
037//      if (oi.getDB1() == true)
038//      {
039//          DriveTrain.F180(Air.getRatio());
040//      }
041//      else if (oi.getDB5() == true)
042//      {
043//          DriveTrain.F90(RobotMap.TurnLeft, Air.getRatio());
044//      }
045//      else if (oi.getDB4() == true)
046//      {
047//          DriveTrain.F90(RobotMap.TurnRight, Air.getRatio());
048//      }
049        if (oi.getGB2() == true)
050        {
051            Jaws.Dump();
052        }
053        else if (oi.getGB3() == true)
054        {
055            Jaws.Load();
056        }
057        else if (oi.getGB2() == false && oi.getGB3() == false)
058        {
059            Jaws.Stop();
060        }
061    }
062
063    protected void TimeSet()
064    {
065        if (Func != RobotMap.DontCare)
066        {
067            if (Func == RobotMap.TurnLeft || Func == RobotMap.TurnRight)
068            {
069                setTimeout(RobotMap.Time90 * Air.getRatio());
070            }
071            else if (Func == RobotMap.TurnRound)
072            {
073                setTimeout(RobotMap.Time180 * Air.getRatio());
074            }
075            else if (Func == RobotMap.TopDump || Func == RobotMap.TopLoad)
076            {
077                setTimeout(0.01);
078            }
079        }
080    }
081
082    // Make this return true when this Command no longer needs to run execute()
083    protected boolean isFinished()
084    {
085        if (Func == RobotMap.DontCare)
086        {
087            return false;
088        }
089        else
090        {
091            return isTimedOut();
092        }
093    }
094
095    // Called once after isFinished returns true
096    protected void end()
097    {
098    }
099
100    // Called when another command which requires one or more of the same
101    // subsystems is scheduled to run
102    protected void interrupted()
103    {
104    }
