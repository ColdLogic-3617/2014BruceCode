005package edu.wpi.first.wpilibj.templates.commands;
006
007import edu.wpi.first.wpilibj.Timer;
008import edu.wpi.first.wpilibj.templates.RobotMap;
009
010/**
011 *
012 * @author Justin, Gabe
013 */
014public class Autonomous extends CommandBase
015{
016    Timer OurTimer = new Timer();
017    boolean Finished = false;
018    double forward, backward, idle;
019
020    public Autonomous()
021    {
022        // Use requires() here to declare subsystem dependencies
023        // eg. requires(chassis);
024        requires(DriveTrain);
025        requires(Air);
026        requires(Jaws);
027        Air.InitShift();
028    }
029
030    // Called just before this Command runs the first time
031    protected void initialize()
032    {
033        Log.println("Autonomous mode initialized.");
034        Log.println("");
035        forward = Shoot.forward;
036        backward = Shoot.backward;
037        idle = Shoot.idle;
038    }
039
040    // Called repeatedly when this Command is scheduled to run
041    protected void execute()
042    {
043        if (RobotMap.AutonMode == "low" || RobotMap.AutonMode == "Low")
044        {
045            lowGoal();      
046        }
047        else if (RobotMap.AutonMode == "high" || RobotMap.AutonMode == "High")
048        {
049            highGoal();
050        }
051        else if (RobotMap.AutonMode == "assist" || RobotMap.AutonMode == "Assist")
052        {
053            assist();
054        }
055    }
056    
057    private void assist()
058    {
059        
060    }
061    
062    private void highGoal()
063    {
064        DriveTrain.GoForward(0.5);
065        Log.println("Moving");
066        OurTimer.delay(1.9);
067        Air.JawOpen();
068        Log.println("Opening jaw.");
069        OurTimer.delay(2);
070        Log.println("Deciding if robot is hobbled.");
071        DriveTrain.Stop();if (RobotMap.Cripple)
072        {
073            Log.println("Robot is hobbled.");
074            forward = forward/8;
075            backward = backward/8;
076        }
077        Log.println("Firing.");
078        Jaws.FlipFire();
079        OurTimer.delay(forward);
080        Log.println("Pausing flipper.");
081        Jaws.FlipStop();
082        OurTimer.delay(idle);
083        Log.println("Retracting flipper.");
084        Jaws.FlipRetract();
085        OurTimer.delay(backward);
086        Log.println("Stopping flipper.");
087        Jaws.FlipStop();
088        DriveTrain.GoBackward(0.5);
089        Log.println("Moving backwards");
090        OurTimer.delay(2);
091        DriveTrain.Stop();
092        if (Air.getGear() ==  false)
093        {
094            DriveTrain.F180(RobotMap.GearRatio, 0.5);
095        }
096        else
097        {
098            DriveTrain.F180(1,0.5);
099        }
100        Log.println("Entering drive configuration.");
101        Air.JawDown();
102        Air.JawClose();
103        Finished = true;
104    }
105    
106    private void lowGoal()
107    {
108        DriveTrain.GoForward(0.5);
109        Log.println("Moving");
110        OurTimer.delay(3);
111        Jaws.setSpeed(1);
112        Log.println("Dumping");
113        OurTimer.delay(1);
114        DriveTrain.Stop();
115        Log.println("Stopped");
116        OurTimer.delay(1);
117        Air.JawDown();
118        Log.println("Lowering");
119        OurTimer.delay(0.7);
120        Air.JawUp();
121        Log.println("Raising Jaw");
122        Jaws.Stop();
123        Log.println("Stopping Loader");
124        OurTimer.delay(0.3);
125        DriveTrain.GoBackward(0.5);
126        Log.println("Moving backwards");
127        OurTimer.delay(2);
128        DriveTrain.Stop();
129        if (Air.getGear() ==  false)
130        {
131            DriveTrain.F180(RobotMap.GearRatio, 0.5);
132        }
133        else
134        {
135            DriveTrain.F180(1,0.5);
136        }
137        Log.println("Entering drive configuration.");
138        Air.JawDown();
139        Air.JawClose();
140        Finished = true;        
141    }
142
143    // Make this return true when this Command no longer needs to run execute()
144    protected boolean isFinished()
145    {
146        return Finished;
147    }
148
149    // Called once after isFinished returns true
150    protected void end()
151    {
152        Log.println("Autonomous finished.");
153        Log.println("");
154    }
155
156    // Called when another command which requires one or more of the same
157    // subsystems is scheduled to run
158    protected void interrupted()
159    {
160        Log.println("Command has been interrupted.");
161    }
162}
