005package edu.wpi.first.wpilibj.templates.commands;
006
007import edu.gappleto.common.ActiveTimer;
008
009import edu.wpi.first.wpilibj.Timer;
010import edu.wpi.first.wpilibj.templates.RobotMap;
011
012/**
013 *
014 * @author gappleto97
015 */
016public class Shoot extends CommandBase 
017{
018    //0.02 seconds ~ 1 "twitch"
019    static double forward = 8 * 0.02,
020            idle = 50 * 0.02, //1 second
021            backward = 75 * 0.02;
022    boolean isFinished = false;
023    
024    public Shoot() 
025    {
026        // Use requires() here to declare subsystem dependencies
027        // eg. requires(chassis);
028        requires(Jaws);
029    }
030
031    // Called just before this Command runs the first time
032    protected void initialize() 
033    {
034        isFinished = false;
035        if (RobotMap.Cripple)
036        {
037            forward = forward/8;
038            backward = backward/8;
039        }
040        Jaws.FlipFire();
041        ActiveTimer.Delay(forward);
042        Jaws.FlipStop();
043        ActiveTimer.Delay(idle);
044        Jaws.FlipRetract();
045        ActiveTimer.Delay(backward);
046        isFinished = true;
047    }
048
049    // Called repeatedly when this Command is scheduled to run
050    protected void execute() 
051    {
052    }
053    
054    public boolean done()
055    {
056        return isFinished;
057    }
058
059    // Make this return true when this Command no longer needs to run execute()
060    protected boolean isFinished() 
061    {
062        return isFinished;
063    }
064
065    // Called once after isFinished returns true
066    protected void end() 
067    {
068    }
069    
070    public double idle()
071    {
072        return idle;
073    }
074    
075    public double forward()
076    {
077        return forward;
078    }
079    
080    public double backward()
081    {
082        return backward;
083    }
084
085    // Called when another command which requires one or more of the same
086    // subsystems is scheduled to run
087    protected void interrupted() 
088    {
089        Log.println("The shooting process has been interrupted.");
090    }
091}
